Rasbian Install

auto lo
iface lo inet loopback

auto eth0
iface eth0 inet static
address 169.254.144.120
netmask 255.255.0.0

auto wlan0
iface wlan0 inet dhcp
wpa-ssid “hnet”
wpa-psk “*“

sudo raspi-config

sudo apt update && sudo apt upgrade -y && sudo reboot
